\subsection*{Purpose   
}\end{DoxyPre}



\begin{DoxyPre}\end{DoxyPre}



\begin{DoxyPre}    DLAMC5 attempts to compute RMAX, the largest machine floating-point   
    number, without overflow.  It assumes that EMAX + \hyperlink{dlamch_8c_a3aa069ac3980707dae1e0530f50d59e4}{abs(EMIN)} sum   
    approximately to a power of 2.  It will fail on machines where this   
    assumption does not hold, for example, the Cyber 205 (EMIN = -28625,\end{DoxyPre}



\begin{DoxyPre}    EMAX = 28718).  It will also fail if the value supplied for EMIN is   
    too large (i.e. too close to zero), probably with overflow.\end{DoxyPre}



\begin{DoxyPre}\subsection*{Arguments   
}\end{DoxyPre}



\begin{DoxyPre}\end{DoxyPre}

            The base of floating-point arithmetic.\end{DoxyPre}



\begin{DoxyPre}    P       (input) INT   
            The number of base BETA digits in the mantissa of a   
            floating-point value.\end{DoxyPre}



\begin{DoxyPre}    EMIN    (input) INT   
            The minimum exponent before (gradual) underflow.\end{DoxyPre}



\begin{DoxyPre}    IEEE    (input) INT   
            \hyperlink{classA}{A} int flag specifying whether or not the arithmetic   
            system is thought to comply with the IEEE standard.\end{DoxyPre}



\begin{DoxyPre}    EMAX    (output) INT   
            The largest exponent before overflow\end{DoxyPre}



\begin{DoxyPre}    RMAX    (output) DOUBLE PRECISION   
            The largest machine floating-point number.   


\end{DoxyPre}



\begin{DoxyPre}\begin{DoxyVerb}   First compute LEXP and UEXP, two powers of 2 that bound   
   abs(EMIN). We then assume that EMAX + abs(EMIN) will sum   
   approximately to the bound that is closest to abs(EMIN).   
   (EMAX is the exponent of the required number RMAX).
\end{DoxyVerb}
