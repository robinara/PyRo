\begin{DoxyPre}
\subsection*{Purpose   
}\end{DoxyPre}



\begin{DoxyPre}\end{DoxyPre}



\begin{DoxyPre}  DLACON estimates the 1-norm of a square matrix \hyperlink{classA}{A}.   
  Reverse communication is used for evaluating matrix-vector products.\end{DoxyPre}



\begin{DoxyPre}\subsection*{Arguments   
}\end{DoxyPre}



\begin{DoxyPre}\end{DoxyPre}



\begin{DoxyPre}  N      (input) INT
         The order of the matrix.  N >= 1.\end{DoxyPre}



\begin{DoxyPre}  V      (workspace) DOUBLE PRECISION array, dimension (N)   
         On the final return, V = A*W,  where  EST = norm(V)/norm(W)   
         (W is not returned).\end{DoxyPre}



\begin{DoxyPre}  X      (input/output) DOUBLE PRECISION array, dimension (N)   
         On an intermediate return, X should be overwritten by   
               \hyperlink{classA}{A} * X,   if KASE=1,   
               \hyperlink{classA}{A}' * X,  if KASE=2,
        and DLACON must be re-called with all the other parameters   
         unchanged.\end{DoxyPre}



\begin{DoxyPre}  ISGN   (workspace) INT array, dimension (N)\end{DoxyPre}



\begin{DoxyPre}  EST    (output) DOUBLE PRECISION   
         An estimate (a lower bound) for norm(A).\end{DoxyPre}



\begin{DoxyPre}  KASE   (input/output) INT
         On the initial call to DLACON, KASE should be 0.   
         On an intermediate return, KASE will be 1 or 2, indicating   
         whether X should be overwritten by \hyperlink{classA}{A} * X  or \hyperlink{classA}{A}' * X.   
         On the final return from DLACON, KASE will again be 0.\end{DoxyPre}
