\begin{DoxyPre}  CLDPERM finds a row permutation so that the matrix has large
  entries on the diagonal.\end{DoxyPre}



\begin{DoxyPre}\subsection*{Arguments
}\end{DoxyPre}



\begin{DoxyPre}\end{DoxyPre}



\begin{DoxyPre}job    (input) int
       Control the action. Possible values for JOB are:
       = 1 : Compute a row permutation of the matrix so that the
             permuted matrix has as many entries on its diagonal as
             possible. The values on the diagonal are of arbitrary size.
             HSL subroutine MC21A/AD is used for this.
       = 2 : Compute a row permutation of the matrix so that the smallest 
             value on the diagonal of the permuted matrix is maximized.
       = 3 : Compute a row permutation of the matrix so that the smallest
             value on the diagonal of the permuted matrix is maximized.
             The algorithm differs from the one used for JOB = 2 and may
             have quite a different performance.
       = 4 : Compute a row permutation of the matrix so that the sum
             of the diagonal entries of the permuted matrix is maximized.
       = 5 : Compute a row permutation of the matrix so that the product
             of the diagonal entries of the permuted matrix is maximized
             and vectors to scale the matrix so that the nonzero diagonal 
             entries of the permuted matrix are one in absolute value and 
             all the off-diagonal entries are less than or equal to one in 
             absolute value.
       Restriction: 1 <= JOB <= 5.\end{DoxyPre}



\begin{DoxyPre}n      (input) int
       The order of the matrix.\end{DoxyPre}



\begin{DoxyPre}nnz    (input) int
       The number of nonzeros in the matrix.\end{DoxyPre}



\begin{DoxyPre}adjncy (input) int*, of size nnz
       The adjacency structure of the matrix, which contains the row
       indices of the nonzeros.\end{DoxyPre}



\begin{DoxyPre}colptr (input) int*, of size n+1
       The pointers to the beginning of each column in ADJNCY.\end{DoxyPre}



\begin{DoxyPre}nzval  (input) complex*, of size nnz
       The nonzero values of the matrix. nzval[k] is the value of
       the entry corresponding to adjncy[k].
       It is not used if job = 1.\end{DoxyPre}



\begin{DoxyPre}perm   (output) int*, of size n
       The permutation vector. perm[i] = j means row i in the
       original matrix is in row j of the permuted matrix.\end{DoxyPre}



\begin{DoxyPre}u      (output) double*, of size n
       If job = 5, the natural logarithms of the row scaling factors.\end{DoxyPre}



\begin{DoxyPre}v      (output) double*, of size n
       If job = 5, the natural logarithms of the column scaling factors. 
       The scaled matrix B has entries b\_ij = a\_ij * exp(u\_i + v\_j).
\end{DoxyPre}
