30 /*
31  * Originally these function objects had no template arguments, e.g.
32  *
33  * struct Add {
34  * template<typename T_numtype1, typename T_numtype2>
35  * static inline BZ_PROMOTE(T_numtype1, T_numtype2)
36  * apply(T_numtype1 a, T_numtype2 b)
37  * { return a + b; }
38  * };
39  *
40  * This made for neater expression templates syntax. However, there are
41  * some situations in which users may want to override type promotion
42  * for certain operations. For example, in theoretical physics, there
43  * are U1 objects which when multiplied yield U1 objects, but when added
44  * yield a different type. To allow for this kind of behaviour, function
45  * objects have been changed to take template parameters:
46  *
47  * template<typename T_numtype1, typename T_numtype2>
48  * struct Add {
49  * typedef BZ_PROMOTE(T_numtype1, T_numtype2) T_numtype;
50  *
51  * static inline T_numtype apply(T_numtype1 a, T_numtype2 b)
52  * { return a + b; }
53  * };
54  *
55  * Type promotion is performed inside the function object. The expression
56  * templates code always looks inside the function object to determine
57  * the type promotion, e.g. Add<int,float>::T_numtype
58  *
59  * Users are free to specialize these function objects for their own types.
60  */
