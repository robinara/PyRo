101 static double double_sum_err(double a, double b, double *err)
102 {
103  double tmp;
104  volatile double c, d, e, g, h, f, x;
105 
106  if (fabs(a) < fabs(b)) {
107  tmp = b;
108  b = a;
109  a = tmp;
110  }
111 
112  c = a + b;
113  e = c - a;
114  g = c - e;
115  h = g - a;
116  f = b - h;
117  d = f - e;
118  x = d + e;
119  if (x != f) {
120  c = a;
121  d = b;
122  }
123  *err = d;
124  return c;
125 }

129 #define _QD_SPLITTER 134217729.0 /* = 2^27 + 1 */

131 static void double_split(double a, double *hi, double *lo)
132 {
133  volatile double b, c;
134  if (a > _QD_SPLIT_THRESH || a < -_QD_SPLIT_THRESH) {
135  a *= 3.7252902984619140625e-09; /* 2^-28 */
136  b = _QD_SPLITTER * a;
137  c = b - a;
138  *hi = b - c;
139  *lo = a - *hi;
140  *hi *= 268435456.0; /* 2^28 */
141  *lo *= 268435456.0; /* 2^28 */
142  } else {
143  b = _QD_SPLITTER * a;
144  c = b - a;
145  *hi = b - c;
146  *lo = a - *hi;
147  }
148 }

151 static double double_mul_err(double a, double b, double *err)
152 {
153  double a_hi, a_lo, b_hi, b_lo;
154  double p = a * b;
155  volatile double c, d;
156  double_split(a, &a_hi, &a_lo);
157  double_split(b, &b_hi, &b_lo);
158  c = a_hi * b_hi - p;
159  d = c + a_hi * b_lo + a_lo * b_hi;
160  *err = d + a_lo * b_lo;
161  return p;
162 }
