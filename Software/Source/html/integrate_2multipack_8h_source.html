13 /* This extension module is a collection of wrapper functions around
14 common FORTRAN code in the packages MINPACK, ODEPACK, and QUADPACK plus
15 some differential algebraic equation solvers.
16 
17 The wrappers are meant to be nearly direct translations between the
18 FORTAN code and Python. Some parameters like sizes do not need to be
19 passed since they are available from the objects.
20 
21 It is anticipated that a pure Python module be written to call these lower
22 level routines and make a simpler user interface. All of the routines define
23 default values for little-used parameters so that even the raw routines are
24 quite useful without a separate wrapper.
25 
26 FORTRAN Outputs that are not either an error indicator or the sought-after
27 results are placed in a dictionary and returned as an optional member of
28 the result tuple when the full_output argument is non-zero.
29 */

143 static PyObject *call_python_function(PyObject *func, npy_intp n, double *x, PyObject *args, int dim, PyObject *error_obj)
144 {
145  /*
146  This is a generic function to call a python function that takes a 1-D
147  sequence as a first argument and optional extra_arguments (should be a
148  zero-length tuple if none desired). The result of the function is
149  returned in a multiarray object.
150  -- build sequence object from values in x.
151  -- add extra arguments (if any) to an argument list.
152  -- call Python callable object
153  -- check if error occurred:
154  if so return NULL
155  -- if no error, place result of Python code into multiarray object.
156  */
157 
158  PyArrayObject *sequence = NULL;
159  PyObject *arglist = NULL;
160  PyObject *arg1 = NULL, *str1 = NULL;
161  PyObject *result = NULL;
162  PyArrayObject *result_array = NULL;
163 
164  /* Build sequence argument from inputs */
165  sequence = (PyArrayObject *)PyArray_SimpleNewFromData(1, &n, NPY_DOUBLE, (char *)x);
166  if (sequence == NULL) PYERR2(error_obj,"Internal failure to make an array of doubles out of first\n argument to function call.");
167 
168  /* Build argument list */
169  if ((arg1 = PyTuple_New(1)) == NULL) {
170  Py_DECREF(sequence);
171  return NULL;
172  }
173  PyTuple_SET_ITEM(arg1, 0, (PyObject *)sequence);
174  /* arg1 now owns sequence reference */
175  if ((arglist = PySequence_Concat( arg1, args)) == NULL)
176  PYERR2(error_obj,"Internal error constructing argument list.");
177 
178  Py_DECREF(arg1); /* arglist has a reference to sequence, now. */
179  arg1=NULL;
180 
181  /* Call function object --- variable passed to routine. Extra
182  arguments are in another passed variable.
183  */
184  if ((result = PyEval_CallObject(func, arglist))==NULL) {
185  goto fail;
186  }
187 
188  if ((result_array = (PyArrayObject *)PyArray_ContiguousFromObject(result, NPY_DOUBLE, dim-1, dim))==NULL)
189  PYERR2(error_obj,"Result from function call is not a proper array of floats.");
190 
191  Py_DECREF(result);
192  Py_DECREF(arglist);
193  return (PyObject *)result_array;
194 
195  fail:
196  Py_XDECREF(arglist);
197  Py_XDECREF(result);
198  Py_XDECREF(arg1);
199  return NULL;
200 }
