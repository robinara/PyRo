77 
78  TinyVector<int, N_destRank + 1> index;
79 
80  // This metaprogram copies elements 0..N-1 of destIndex into index
81  _bz_meta_vecAssign<N_index, 0>::assign(index, destIndex,
82  _bz_update<int,int>());

84  int lbound = iter_.lbound(N_index);
85  int ubound = iter_.ubound(N_index);

87  // NEEDS_WORK: replace with tiny(int()) and huge(int()) once
88  // <limits> widely available
89  BZPRECHECK((lbound != INT_MIN) && (ubound != INT_MAX),
90  "Array reduction performed over rank " << N_index
91  << " is unbounded." << endl
92  << "There must be an array object in the expression being reduced"
93  << endl << "which provides a bound in rank " << N_index << ".");
94 
95  reduce_.reset();
96 
97  for (index[N_index] = iter_.lbound(N_index);
98  index[N_index] <= ubound; ++index[N_index])
99  {
100  if (!reduce_(iter_(index), index[N_index]))
101  break;
102  }
103 
104  return reduce_.result(ubound-lbound+1);
105  }

210 // method for properly initializing the ordering values
211  void computeOrdering()
212  {
213  TinyVector<bool,rank> in_ordering;
214  in_ordering = false;
215 
216  int j = 0;
217  for (int i=0; i<rank; ++i)
218  {
219  int orderingj = iter_.ordering(i);
220  if (orderingj != INT_MIN && orderingj < rank &&
221  !in_ordering(orderingj)) { // unique value in ordering array
222  in_ordering(orderingj) = true;
223  ordering_(j++) = orderingj;
224  }
225 
226  }
227 
228  // It is possible that ordering is not a permutation of 0,...,rank-1.
229  // In that case j will be less than rank. We fill in ordering with
230  // the unused values in decreasing order.
231  for (int i = rank-1; j < rank; ++j) {
232  while (in_ordering(i))
233  --i;
234  ordering_(j) = i--;
235  }
236  }
