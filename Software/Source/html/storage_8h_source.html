22 /*
23  * Declaration of class GeneralStorage<N_rank>
24  *
25  * This class describes a storage format for an N-dimensional array.
26  * The dimensions can be stored in an arbitrary order (for example, as
27  * a C-style row major array or Fortran-style column major array, or
28  * something else entirely). Each dimension can be stored in either
29  * ascending (the most common) or descending order. Each dimension
30  * can have its own base (starting index value: e.g. 0 for C-style arrays,
31  * 1 for Fortran arrays).
32  *
33  * GeneralArrayStorage<N> defaults to C-style arrays. To implement
34  * other storage formats, subclass and modify the constructor. The
35  * class FortranArray, below, is an example.
36  *
37  * Objects inheriting from GeneralArrayStorage<N> can be passed as
38  * an optional constructor argument to Array objects.
39  * e.g. Array<int,3> A(16,16,16, FortranArray<3>());
40  * will create a 3-dimensional 16x16x16 Fortran-style array.
41  */

132  /*
133  * ordering_[] specifies the order in which the array is stored in
134  * memory. For a newly allocated array, ordering_(0) will give the
135  * rank with unit stride, and ordering_(N_rank-1) will be the rank
136  * with largest stride. An order like [2, 1, 0] corresponds to
137  * C-style array storage; an order like [0, 1, 2] corresponds to
138  * Fortran array storage.
139  *
140  * ascendingFlag_[] indicates whether the data in a rank is stored
141  * in ascending or descending order. Most of the time these values
142  * will all be true (indicating ascending order). Some peculiar
143  * formats (e.g. MS-Windows BMP image format) store the data in
144  * descending order.
145  *
146  * base_[] gives the first valid index for each rank. For a C-style
147  * array, all the base_ elements will be zero; for a Fortran-style
148  * array, they will be one. base_[] can be set arbitrarily using
149  * the Array constructor which takes a Range argument, e.g.
150  * Array<float,2> A(Range(30,40),Range(23,33));
151  * will create an array with base_[] = { 30, 23 }.
152  */
153  TinyVector<int, N_rank> ordering_;
154  TinyVector<bool, N_rank> ascendingFlag_;
155  TinyVector<int, N_rank> base_;
