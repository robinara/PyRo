224 struct bounds {
225  static int compute_ascending(int BZ_DEBUG_PARAM(rank),
226  int ascending1, int ascending2)
227  {
228  // The value INT_MIN indicates that there are no arrays
229  // in a subtree of the expression. This logic returns
230  // whichever ascending is available. If there are two
231  // conflicting ascending values, this is an error.
232 
233  if (ascending1 == ascending2)
234  return ascending1;
235  else if (ascending1 == INT_MIN)
236  return ascending2;
237  else if (ascending2 == INT_MIN)
238  return ascending1;
239 
240  BZ_DEBUG_MESSAGE("Two array operands have different"
241  << endl << "ascending flags: for rank " << rank
242  << ", the flags are " << ascending1 << " and "
243  << ascending2 << endl);
244  BZ_PRE_FAIL;
245  return 0;
246  }
247 
248  static int compute_ordering(int BZ_DEBUG_PARAM(rank),
249  int order1, int order2)
250  {
251  // The value INT_MIN indicates that there are no arrays
252  // in a subtree of the expression. This logic returns
253  // whichever ordering is available. If there are two
254  // conflicting ordering values, this is an error.
255 
256  if (order1 == order2)
257  return order1;
258  else if (order1 == INT_MIN)
259  return order2;
260  else if (order2 == INT_MIN)
261  return order1;
262 
263  BZ_DEBUG_MESSAGE("Two array operands have different"
264  << endl << "orders: for rank " << rank << ", the orders are "
265  << order1 << " and " << order2 << endl);
266  BZ_PRE_FAIL;
267  return 0;
268  }
