20 /*
21  * These numeric inquiry functions are provided as an alternative
22  * to the somewhat klunky numeric_limits<T>::yadda_yadda syntax.
23  * Where a similar Fortran 90 function exists, the same name has
24  * been used.
25  *
26  * The argument in all cases is a dummy of the appropriate type
27  * (double, int, etc.)
28  *
29  * These functions assume that numeric_limits<T> has been specialized
30  * for the appropriate case. If not, the results are not useful.
31  */

52 /*
53  * This traits class provides zero and one values for numeric
54  * types. This was previously a template function with specializations,
55  * but the specializations were causing multiply-defined symbols
56  * at link time. TV 980226
57  */
58 
59 template<typename T_numtype>
60 struct _bz_OneZeroTraits {
61  static inline T_numtype zero() { return 0; }
62  static inline T_numtype one() { return 1; }
63 };
