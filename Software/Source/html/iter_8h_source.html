174 template<typename T, int N>
175 ConstArrayIterator<T,N>& ConstArrayIterator<T,N>::operator++()
176 {
177  BZPRECHECK(data_ != 0, "Attempted to iterate past the end of an array.");
178 
179  data_ += stride_;
180 
181  if (data_ != last_[0])
182  {
183  // We hit this case almost all the time.
184  ++pos_[maxRank_];
185  return *this;
186  }
187 
188  // We've hit the end of a row/column/whatever. Need to
189  // increment one of the loops over another dimension.
190 
191  int j = 1;
192  for (; j < N; ++j)
193  {
194  int r = order_(j);
195  data_ = const_cast<T*>(stack_[j]);
196  data_ += strides_[r];
197  ++pos_(r);
198 
199  if (data_ != last_[j])
200  break;
201  }
202 
203  // All done?
204  if (j == N)
205  {
206  // Setting data_ to 0 indicates the end of the array has
207  // been reached, and will match the end iterator.
208  data_ = 0;
209  return *this;
210  }
211 
212  stack_[j] = data_;
213 
214  // Now reset all the last pointers
215  for (--j; j >= 0; --j)
216  {
217  int r2 = order_(j);
218  stack_[j] = data_;
219  last_[j] = data_ + extent_(r2) * strides_(r2);
220  pos_(r2) = lbound_(r2);
221  }
222 
223  return *this;
224 }
