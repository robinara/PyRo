80 /*
81  * Numpy basic types codes correspond to C basic types, but they remain
82  * different even if the corresponding types have the same size.
83  *
84  * Most commonly: int and long, so normalize to long.
85  */
86 static NPY_INLINE
87 int NI_NormalizeType(int type_num)
88 {
89 #if NPY_SIZEOF_INT == NPY_SIZEOF_LONG
90  if (type_num == NPY_INT) type_num = NPY_LONG;
91  if (type_num == NPY_UINT) type_num = NPY_ULONG;
92 #endif
93  return type_num;
94 }

322 /* Move the pointer to the filter offsets according to the given
323  coordinates: */
324 #define NI_FILTER_GOTO(iteratorf, iterator, fbase, pointerf) \
325 { \
326  int _ii; \
327  npy_intp _jj; \
328  pointerf = fbase; \
329  for(_ii = iterator.rank_m1; _ii >= 0; _ii--) { \
330  npy_intp _pp = iterator.coordinates[_ii]; \
331  npy_intp b1 = (iteratorf).bound1[_ii]; \
332  npy_intp b2 = (iteratorf).bound2[_ii]; \
333  if (_pp < b1) { \
334  _jj = _pp; \

336  _jj = _pp + b1 - b2; \
337  } else { \

339  } \
