46  /*
47  * In testsuite mode, these routines allow a test suite to check
48  * that precondition checking is being done properly. A typical
49  * use looks like this:
50  *
51  * beginCheckAssert();
52  * // Some operation which should cause an assert to fail
53  * endCheckAssert();
54  *
55  * The routine beginCheckAssert() sets a flag which results in
56  * failed asserts being silently tallied. If no asserts have
57  * failed by the time endCheckAssert() is invoked, the program
58  * halts and issues an error code.
59  *
60  * In normal operation (i.e. when beginCheckAssert() has not
61  * been called), failed preconditions will cause the program
62  * to halt and issue an error code. -- TV 980226
63  */
64 
65  inline void checkAssert(bool condition, const char* where=0,
66  int line=0)
67  {
68  if (assertFailMode == true)
69  {
70  if (condition == true)
71  ++assertSuccessCount;
72  else
73  ++assertFailCount;
74  }
75  else {
76  if (!condition)
77  {
78  cerr << "Unexpected assert failure!" << endl;
79  if (where)
80  cerr << where << ":" << line << endl;
81  cerr.flush();
82  assert(0);
83  }
84  }
85  }
86 
87  inline void beginCheckAssert()
88  {
89  assertFailMode = true;
90  assertSuccessCount = 0;
91  assertFailCount = 0;
92  }
93 
94  inline void endCheckAssert()
95  {
96  assert(assertFailMode == true);
97  assertFailMode = false;
98  if (assertFailCount == 0)
99  {
100  cerr << "Assert check failed!" << endl;
101  assert(0);
102  }
103  }
