336 // cbrt(x) Cube root
337 // ceil(d), ceill(ld) Smallest f-int not less than x
338 // int class(d) Classification of x (FP_XXXXX)
339 // cos(d), cosl(ld) Cosine
340 // cosh(d), coshl(ld) Hyperbolic cosine
341 // copysign(d,d) Return 1st arg with same sign as 2nd
342 // drem(x,x) IEEE remainder
343 // exp(d), expl(ld) Exponential
344 // expm1(d) Exp(x)-1
345 // erf(d), erfl(ld) Error function
346 // erfc(d), erfcl(ld) Complementary error function
347 // fabs(d), fabsl(ld) Floating point absolute value
348 // int finite(d) Nonzero if finite
349 // floor(d), floor(ld) Largest f-int not greater than x
350 // fmod(d,d), fmodl(ld,ld) Floating point remainder
351 // frexp(d, int* e) Break into mantissa/exponent (*)
352 // frexpl(ld, int* e) Break into mantissa/exponent (*)
353 // gammaFunc(d) Gamma function (** needs special
354 // implementation using lgamma)
355 // hypot(d,d) Hypotenuse: sqrt(x*x+y*y)
356 // int ilogb(d) Integer unbiased exponent
357 // int isnan(d) Nonzero if NaNS or NaNQ
358 // int itrunc(d) Truncate and convert to integer
359 // j0(d) Bessel function first kind, order 0
360 // j1(d) Bessel function first kind, order 1
361 // jn(int, double) Bessel function first kind, order i
362 // ldexp(d,i), ldexpl(ld,i) Compute d * 2^i
363 // lgamma(d), lgammald(ld) Log absolute gamma
364 // log(d), logl(ld) Natural logarithm
365 // logb(d) Unbiased exponent (IEEE)
366 // log1p(d) Compute log(1 + x)
