90 /* satisfies ensures that 'a' meets a set of requirements and matches
91 the specified type.
92 */
93 static int
94 satisfies(PyArrayObject *a, int requirements, NumarrayType t)
95 {
96  int type_ok = (t == tAny) || PyArray_EquivTypenums(a->descr->type_num, t);
97 
98  if (PyArray_ISCARRAY(a))
99  return type_ok;
100  if (PyArray_ISBYTESWAPPED(a) && (requirements & NPY_NOTSWAPPED))
101  return 0;
102  if (!PyArray_ISALIGNED(a) && (requirements & NPY_ALIGNED))
103  return 0;
104  if (!PyArray_ISCONTIGUOUS(a) && (requirements & NPY_CONTIGUOUS))
105  return 0;
106  if (!PyArray_ISWRITEABLE(a) && (requirements & NPY_WRITEABLE))
107  return 0;
108  if (requirements & NPY_ENSURECOPY)
109  return 0;
110  return type_ok;
111 }

113 static PyArrayObject *
114 NA_OutputArray(PyObject *a, NumarrayType t, int requires)
115 {
116  PyArray_Descr *dtype;
117  PyArrayObject *ret;
118 
119  if (!PyArray_Check(a) || !PyArray_ISWRITEABLE(a)) {
120  PyErr_Format(PyExc_TypeError,
121  "NA_OutputArray: only writeable arrays work for output.");
122  return NULL;
123  }
124 
125  if (satisfies((PyArrayObject *)a, requires, t)) {
126  Py_INCREF(a);
127  return (PyArrayObject *)a;
128  }
129  if (t == tAny) {
130  dtype = PyArray_DESCR(a);
131  Py_INCREF(dtype);
132  }
133  else {
134  dtype = PyArray_DescrFromType(t);
135  }
136  ret = (PyArrayObject *)PyArray_Empty(PyArray_NDIM(a), PyArray_DIMS(a),
137  dtype, 0);
138  ret->flags |= NPY_UPDATEIFCOPY;
139  ret->base = a;
140  PyArray_FLAGS(a) &= ~NPY_WRITEABLE;
141  Py_INCREF(a);
142  return ret;
143 }

145 /* NA_IoArray is a combination of NA_InputArray and NA_OutputArray.
146 
147 Unlike NA_OutputArray, if a temporary is required it is initialized to a copy
148 of the input array.
149 
150 Unlike NA_InputArray, deallocating any resulting temporary array results in a
151 copy from the temporary back to the original.
152 */
153 static PyArrayObject *
154 NA_IoArray(PyObject *a, NumarrayType t, int requires)
155 {
156  PyArrayObject *shadow = NA_InputArray(a, t, requires | NPY_UPDATEIFCOPY );
157 
158  if (!shadow) return NULL;
159 
160  /* Guard against non-writable, but otherwise satisfying requires.
161  In this case, shadow == a.
162  */
163  if (!PyArray_ISWRITEABLE(shadow)) {
164  PyErr_Format(PyExc_TypeError,
165  "NA_IoArray: I/O array must be writable array");
166  PyArray_XDECREF_ERR(shadow);
167  return NULL;
168  }
169 
170  return shadow;
171 }
