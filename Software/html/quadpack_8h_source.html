12 /* This extension module is a collection of wrapper functions around
13 common FORTRAN code in the packages MINPACK, ODEPACK, and QUADPACK plus
14 some differential algebraic equation solvers.
15 
16 The wrappers are meant to be nearly direct translations between the
17 FORTAN code and Python. Some parameters like sizes do not need to be
18 passed since they are available from the objects.
19 
20 It is anticipated that a pure Python module be written to call these lower
21 level routines and make a simpler user interface. All of the routines define
22 default values for little-used parameters so that even the raw routines are
23 quite useful without a separate wrapper.
24 
25 FORTRAN Outputs that are not either an error indicator or the sought-after
26 results are placed in a dictionary and returned as an optional member of
27 the result tuple when the full_output argument is non-zero.
28 */

72  store->global0 = (void *)quadpack_python_function;
73  store->global1 = (void *)quadpack_extra_arguments;
74  memcpy(&(store->jmp), &quadpack_jmpbuf, sizeof(jmp_buf));
75  store->arg = arg;
76  if (store->arg == NULL) {
77  if ((store->arg = PyTuple_New(0)) == NULL)

82  }
83  if (!PyTuple_Check(store->arg)) {

85  Py_XDECREF(store->arg);

87  }
