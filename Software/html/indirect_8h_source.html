86 // Specialization for RectDomain<N>
87 template<typename T_array, typename T_arrayiter, int N_rank, typename T_expr>
88 inline void applyOverSubdomain(const T_array& BZ_DEBUG_PARAM(array), T_arrayiter& arrayIter,
89  RectDomain<N_rank> subdomain,
90  T_expr expr)
91 {
92  typedef _bz_typename T_array::T_numtype T_numtype;
93 
94  // Assume that the RectDomain<N_rank> is a 1-D strip.
95  // Find the dimension in which the strip is oriented. This
96  // variable is static so that we cache the value; likely to be
97  // the same for all strips within a container.
98 
99  static int stripDim = 0;
100 
101  if (subdomain.lbound(stripDim) == subdomain.ubound(stripDim))
102  {
103  // Cached value was wrong, find the correct value of stripDim
104  for (stripDim=0; stripDim < N_rank; ++stripDim)
105  if (subdomain.lbound(stripDim) != subdomain.ubound(stripDim))
106  break;
107 
108  // Handle case where the strip is just a single point
109  if (stripDim == N_rank)
110  stripDim = 0;
111  }
112 
113 #ifdef BZ_DEBUG
114  // Check that this is in fact a 1D strip
115  for (int i=0; i < N_rank; ++i)
116  if ((i != stripDim) && (subdomain.lbound(i) != subdomain.ubound(i)))
117  BZPRECHECK(0, "In indirection using an STL container of RectDomain<"
118  << N_rank << ">, one of" << endl << "the RectDomain objects was not"
119  " a one-dimensional strip:" << endl << "RectDomain<" << N_rank
120  << ">::lbound() = " << subdomain.lbound() << endl
121  << "RectDomain<" << N_rank << ">::ubound() = " << subdomain.ubound())
122 #endif
123 
124  // Check that the start and end position are in range
125  BZPRECHECK(array.isInRange(subdomain.lbound()),
126  "In indirection using an STL container of RectDomain<"
127  << N_rank << ">, one of" << endl << "the RectDomain objects has a"
128  " lbound which is out of range:" << endl
129  << subdomain.lbound() << endl
130  << "Array lower bounds: " << array.lbound() << endl
131  << "Array upper bounds: " << array.ubound() << endl)
132 
133  BZPRECHECK(array.isInRange(subdomain.ubound()),
134  "In indirection using an STL container of RectDomain<"
135  << N_rank << ">, one of" << endl << "the RectDomain objects has a"
136  " ubound which is out of range:" << endl
137  << subdomain.lbound() << endl
138  << "Array lower bounds: " << array.lbound() << endl
139  << "Array upper bounds: " << array.ubound() << endl)
140 
141  // Position at the beginning of the strip
142  arrayIter.moveTo(subdomain.lbound());
143  expr.moveTo(subdomain.lbound());
144 
145  // Loop through the strip
146 
147 #ifdef BZ_USE_FAST_READ_ARRAY_EXPR
148 
149  bool useUnitStride = arrayIter.isUnitStride(stripDim)
150  && expr.isUnitStride(stripDim);
151 
152  int lbound = subdomain.lbound(stripDim);
153  int ubound = subdomain.ubound(stripDim);
154 
155  if (useUnitStride)
156  {
157  T_numtype* restrict data = const_cast<T_numtype*>(arrayIter.data());
158 
159  int length = ubound - lbound + 1;
160  for (int i=0; i < length; ++i)
161  *data++ = expr.fastRead(i);
162  }
163  else {
164 #endif
165 
166  arrayIter.loadStride(stripDim);
167  expr.loadStride(stripDim);
168 
169  for (int i=lbound; i <= ubound; ++i)
170  {
171  *const_cast<_bz_typename T_arrayiter::T_numtype*>(arrayIter.data())
172  = *expr;
173  expr.advance();
174  arrayIter.advance();
175  }
176 
177 #ifdef BZ_USE_FAST_READ_ARRAY_EXPR
178  }
179 #endif
180 }

370 // The cp_traits class handles promotion of singleton integers to
371 // containers. It takes two template parameters:
372 // T = argument type
373 // T2 = container type
374 // If T is an integer, then a container of type T2 is created and the
375 // integer is inserted. This container is returned.
376 // Otherwise, T is assumed to be the same type as T2, and the original
377 // container is returned.
378 
379 template<typename T, typename T2>
380 struct cp_traits {
381  typedef T T_container;
382 
383  static const T_container& make(const T& x)
384  { return x; }
385 };
386 
387 template<typename T2>
388 struct cp_traits<int,T2> {
389  typedef T2 T_container;
390 
391  static T2 make(int x)
392  {
393  T2 singleton;
394  singleton.push_back(x);
395  return singleton;
396  }
397 };
